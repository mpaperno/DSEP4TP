54 //! Permission flags returned by `permissions()` and used with `setPermissions()`. These conform to GNU/Node/etc file mode constants,
55 //! with the addition of "OWN" (owner) bits. On NTFS the owner may differ from the user. On Unix/POSIX the "USR" bits return owner information as per convention.
56 //! \note Because of differences in the platforms, the semantics of S_IRUSR, S_IWUSR and S_IXUSR are platform-dependent:
57 //! On Unix, the rights of the owner of the file are returned and on Windows the rights of the current user are returned.
58 enum Permission : quint16 {
59 S_IROWN = QFileDevice::ReadOwner, //!< The file is readable by the owner of the file.
60 S_IWOWN = QFileDevice::WriteOwner, //!< The file is writable by the owner of the file.
61 S_IXOWN = QFileDevice::ExeOwner, //!< The file is executable by the owner of the file.
62 S_IRWXN = S_IROWN | S_IWOWN | S_IXOWN, //!< S_IROWN | S_IWOWN | S_IXOWN
63 S_IRUSR = QFileDevice::ReadUser, //!< The file is readable by the user.
64 S_IWUSR = QFileDevice::WriteUser, //!< The file is writable by the user.
65 S_IXUSR = QFileDevice::ExeUser, //!< The file is executable by the user.
66 S_IRWXU = S_IRUSR | S_IWUSR | S_IXUSR, //!< S_IRUSR | S_IWUSR | S_IXUSR
67 S_IRGRP = QFileDevice::ReadGroup, //!< The file is readable by the group.
68 S_IWGRP = QFileDevice::WriteGroup, //!< The file is writable by the group.
69 S_IXGRP = QFileDevice::ExeGroup, //!< The file is executable by the group.
70 S_IRWXG = S_IRGRP | S_IWGRP | S_IXGRP, //!< S_IRGRP | S_IWGRP | S_IXGRP
71 S_IROTH = QFileDevice::ReadOther, //!< The file is readable by anyone.
72 S_IWOTH = QFileDevice::WriteOther, //!< The file is writable by anyone.
73 S_IXOTH = QFileDevice::ExeOther, //!< The file is executable by anyone.
74 S_IRWXO = S_IROTH | S_IWOTH | S_IXOTH, //!< S_IROTH | S_IWOTH | S_IXOTH
75 };
