37//! \ingroup FileSystem
38//! The `Dir` class provides static functions for directory operations, which are accessed in JS with the `Dir.` qualifier.
39//! These are "atomic" operations such as getting information about a path, manipulating directories and so forth.
40class Dir : public QObject
41{
42 Q_OBJECT
43 explicit Dir(bool isStatic) : Dir()
44 {
45 if (isStatic)
46 QQmlEngine::setObjectOwnership(this, QQmlEngine::CppOwnership);
47 }
48
49 public:
50 using QObject::QObject;
51
52 static Dir *instance()
53 {
54 static Dir instance(true);
55 return &instance;
56 }
57
58 explicit Dir(QObject *p = nullptr) : QObject(p)
59 {
60 setObjectName("Dir");
61 }
62
63 // Dir helpers
64 //! \{
65
66 //! Creates the directory path `path`. The function will create all parent directories necessary to create the directory.
67 //! Returns `true` if successful; otherwise returns `false`. If the path already exists when this function is called, it will return `true`.
68 Q_INVOKABLE static bool mkpath(const QString &path) { return QDir().mkpath(path); }
69 //! Creates a sub-directory called `dirName`. Returns `true` on success; otherwise returns `false`.
70 //! If the directory already exists when this function is called, it will return `false`.
71 Q_INVOKABLE static bool mkdir(const QString &dirName) { return QDir().mkdir(dirName); }
72 //! Removes the directory path `path`. The function will remove all parent directories in `path`, provided that they are empty. This is the opposite of `mkpath()`.
73 //! Returns `true` if successful; otherwise returns `false`.
74 Q_INVOKABLE static bool rmpath(const QString &path) { return QDir().rmpath(path); }
75 //! This function has two modes of operation:
76 //! * If `recurse` = `false` (default): Removes the directory specified by `dirName`. The directory must be empty for `rmdir()` to succeed.
77 //! * If `recurse` = `true`: Removes the directory specified by `dirName`, **including all its contents**. If a file or directory cannot be removed it keeps going and attempts to
78 //! delete as many files and sub-directories as possible, then returns `false`. If the directory was already removed, the method returns `true` (expected result already reached).
79 //! Returns `true` if successful; otherwise returns `false`, except as noted above.
80 Q_INVOKABLE static bool rmdir(const QString &dirName, bool recurse = false) { return recurse ? QDir(dirName).removeRecursively() : QDir().rmdir(dirName); }
81
82 //! \}
83 // Dir info
84 //! \{
85
86 //! Returns `true` if `path` exists in the file system, `false` otherwise.
87 Q_INVOKABLE static bool exists(const QString &path) { return QDir(path).exists(); }
88 //! Returns `true` if `path` is absolute (that is, from a root directory), `false` otherwise (if it relative).
89 Q_INVOKABLE static bool isAbs(const QString &path) { return QDir::isAbsolutePath(path); }
90
91 //! \}
92 // Paths
93 //! \{
94
95 //! Returns the absolute path of the application's current directory. The current directory is the directory at which this application was started at by the parent process.
96 Q_INVOKABLE static QString cwd() { return QDir::currentPath(); }
97 //! Returns the absolute path of the user's home directory. (Under Windows this function will return the directory of the current user's profile, eg `C:/Users/Username`)
98 //! Under non-Windows operating systems the HOME environment variable is used if it exists, otherwise the path returned by the rootPath().
99 Q_INVOKABLE static QString home() { return QDir::homePath(); }
100 //! Returns the absolute canonical path of the system's temporary directory. On Unix/Linux systems this is the path in the TMPDIR environment variable or /tmp if TMPDIR is not defined.
101 //! On Windows this is usually the path in the TEMP or TMP environment variable. The path returned by this method doesn't end with a directory separator unless it is the root directory (of a drive).
102 Q_INVOKABLE static QString temp() { return QDir::tempPath(); }
103 //! Returns the absolute path of the root directory. For Windows file systems this normally returns the boot drive letter (typically "c:/"). For Unix/Mac operating systems this returns "/".
104 Q_INVOKABLE static QString root() { return QDir::rootPath(); }
105
106 //! Returns the native directory separator: "/" under Unix and "\" under Windows. Note: You do not need to use this function to build file paths. You can always use "/" and it will be adjusted if needed.
107 Q_INVOKABLE static QString separator() { return QDir::separator(); }
108 //! Returns pathName with the '/' separators converted to separators that are appropriate for the underlying operating system.
109 Q_INVOKABLE static QString toNative(const QString &path) { return QDir::toNativeSeparators(path); }
110 //! Returns pathName using '/' as file separator.
111 Q_INVOKABLE static QString fromNative(const QString &path) { return QDir::fromNativeSeparators(path); }
112 //! Returns path with directory separators normalized (that is, platform-native separators converted to "/") and redundant ones removed, and "."s and ".."s resolved (as far as possible). Symbolic links are kept
113 Q_INVOKABLE static QString clean(const QString &path) { return QDir::cleanPath(path); }
114 //! Returns the absolute path (a path that starts with "/" or with a drive specification), which may contain symbolic links, but never contains redundant ".", ".." or multiple separators.
115 Q_INVOKABLE static QString abs(const QString &path) { return QDir(path).absolutePath(); }
116 //! Returns the canonical path, i.e. a path without symbolic links or redundant "." or ".." elements.
117 Q_INVOKABLE static QString normalize(const QString &path) { return QDir(path).canonicalPath(); }
118
119 //! \}
120};
