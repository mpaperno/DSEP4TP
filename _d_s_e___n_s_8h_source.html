55//! Script instance saved default value type. These values determine what happens when a script instance is restored from persistent storage.
56//! \sa DynamicScript.defaultType
57enum SavedDefaultType : quint8 {
58 NoSavedDefault, //!< The instance is not saved in persistent settings, default value type is not applicatble.
59 FixedValueDefault, //!< Instance is created with a fixed default or empty value (specified in `DynamicScript.defaultValue`)
60 CustomExprDefault, //!< Instance is created with default value coming from evaluating a custom expression (specified in `DynamicScript.defaultValue`)
61 LastExprDefault, //!< Instance is created with default value coming from evaluating the last saved primary expression
62 //! (exactly as last sent from the corresponding Touch Portal action/connector).
63};
64Q_ENUM_NS(SavedDefaultType)

66//! Defines how an action should behave when it is being "activated" as in the case of a button.
67//! A button technically how two transition states -- when pressed and when released. The time between those events
68//! can also either be ignored, reacting only to the transition(s), or used to do something,
69//! like repeat some command over and over. An action may also invoke different functions on initial press vs. on release.
70//! \sa DynamicScript.activation
71enum ActivationBehavior : quint8 {
72 NoActivation = 0x00, //!< The action is never activated, evaluation never happens.
73 OnPress = 0x01, //!< Evaluation happens on initial activation (eg: button press).
74 OnRelease = 0x02, //!< Evaluation happens on de-activation (eg: button release).
75 RepeatOnHold = 0x04, //!< Evaluation happens repeatedly while the action is active (eg: button is held down).
76};
77Q_FLAG_NS(ActivationBehavior)
78//! The `DynamicScript.ActivationBehaviors` type stores an OR combination of `DynamicScript.ActivationBehavior` values.
79Q_DECLARE_FLAGS(ActivationBehaviors, ActivationBehavior)

81//! Action repeat property type. \sa DSE.setActionRepeat(), DSE.adjustActionRepeat()
82enum RepeatProperty : quint8 {
83 RepeatRateProperty = 0x01, //!< Rate, or pause interval between repetitions, in milliseconds.
84 RepeatDelayProperty = 0x02, //!< Initial delay before the the first repetition is activated, in milliseconds.
85 AllRepeatProperties = RepeatRateProperty | RepeatDelayProperty //!< OR combination of Rate and Delay properties.
86};
87Q_ENUM_NS(RepeatProperty)
