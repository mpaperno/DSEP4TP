107 //! Reads up to `maxLines` lines from `file` starting at `fromLine` (default = 0, start of file) and returns the contents as a string. Can search from start or end of file.
108 //! \n `maxLines` can be `0` in which case all remaining lines in the file are returned. This is useful when `fromLine != 0` to skip a number of lines
109 //! and then return the rest.
110 //! \n `fromLine` can be negative, in which case the file will be read from the end, backwards. In this case `-1` means starting at the end of the file,
111 //! `-2` means to skip one line ("start at second-to-last line"), and so on.
112 //! \n Set `trimTrailingNewlines` to `true` (default) to trim/skip over any empty lines from the end of the file. When searching backwards,
113 //! setting `trimTrailingNewlines` to `false` will count every newline from the end, including any potential newline at end of the last line of the file.
114 //! \note Note that if no newlines are found in the file then the full contents will be returned (a potentially expensive operation).
115 //! This operation works on all text files regardless of line endings.
116 Q_INVOKABLE QString readLines(const QString &file, int maxLines, int fromLine = 0, bool trimTrailingNewlines = true)
117 {
118 if (maxLines < 0)
119 return QString();
120 QFile fh(file);
121 if (!open_impl(fh, qjsEngine(this), O_RDONLY /*| O_TEXT*/))
122 return QString();
123 if (fh.size() < 2) {
124 fh.close();
125 return QString();
126 }
127 QString ret;
128 if (fromLine >= 0) {
129 ret = readLinesFromStart(fh, maxLines, fromLine, trimTrailingNewlines);
130 }
131 else {
132 fh.seek(fh.size() - 1);
133 ret = readLinesFromEnd(fh, maxLines, fromLine, trimTrailingNewlines);
134 }
135 fh.close();
136 //qCDebug(lcPlugin) << ret.toUtf8().toHex(':');
137 return ret;
138 }

141 //! Writes `data` to `file` and returns the number of bytes written. Returns `-1` on error. `mode` flags can be a combination of `FileLib::OpenModeFlag` enums:
142 //! * `FS.O_BIN` write in binary mode (default is to write as text).
143 //! * `FS.O_APPEND` to append to file (default is to truncate/replace it).
144 //! * `FS.O_EXCL` to fail if the file exists, or `FS.O_NOCREAT` to to fail if the file _doesn't_ already exist (both will return an error if the condition isn't met).
145 //! Throws an `Error` type object if file loading fails (file not found, `FS.O_EXCL` or `FS.O_NOCREAT` checks failed, etc).
146 Q_INVOKABLE qint64 write(const QString &file, const QByteArray &data, ScriptLib::FS::OpenMode mode = O_TEXT) const
147 {
148 qint64 ret;
149 QFile fh(file);
150 if (fh.open(toQfileFlags(mode.setFlag(O_RDONLY, false).setFlag(O_WRONLY)))) {
151 ret = fh.write(data);
152 fh.close();
153 }
154 else {
155 ret = -1;
156 if (QJSEngine *jse = qjsEngine(this))
157 jse->throwError(QJSValue::GenericError, "Could not write to file '" + file + "': " + fh.errorString());
158 else
159 qCWarning(lcPlugin) << "Could not write file" << file << ":" << fh.errorString();
160 }
161 return ret;
162 }

266 // Truncates str to exclude trailing newline(s) including any CRs.
267 static QString &trimTrailingNewlines(QString &str)
268 {
269 for (int n = str.size() - 1; n > -1; --n) {
270 const QChar &ch = str.at(n);
271 if (ch != '\n' && ch != '\r') {
272 str.truncate(n+1);
273 break;
274 }
275 }
276 return str;
277 }

279 static QString readLinesFromStart(QFile &fh, int maxLines, int fromLine, bool trimTrailing)
280 {
281 // Seek forward
282 int count = 0;
283 if (fromLine > 0) {
284 // Skip lines
285 const qint64 len = fh.size() - fh.pos();
286 int p = 0;
287 char ch;
288 for ( ; p < len; ++p)
289 if (fh.read(&ch, 1) != 1 || (ch == '\n' && ++count == fromLine))
290 break;
291 if (p >= len)
292 return QString();
293 count = 0;
294 }
295 QString ret;
296 // If !maxLines then returns the rest of the file from current position, otherwise add line-by-line.
297 if (maxLines) {
298 QTextStream strm(&ret, QIODevice::WriteOnly);
299 while (count < maxLines && !fh.atEnd()) {
300 const QString l = fh.readLine();
301 strm << l;
302 // end or error is signaled by empty return or lack of nl
303 if (l.isEmpty() || !l.endsWith('\n'))
304 break;
305 ++count;
306 }
307 }
308 else {
309 ret = fh.readAll();
310 }
311 return trimTrailing ? trimTrailingNewlines(ret) : ret;
312 }
313
314 static QString readLinesFromEnd(QFile &fh, int maxLines, int fromLine, bool trimTrailing)
315 {
316 // Seek backwards
317 int count = 0;
318 int skipLines = (1 + fromLine) * -1;
319 qint64 p = fh.pos();
320 if (p < 1)
321 return QString();
322 char ch;
323
324 // ignore trailing newlines
325 if (trimTrailing) {
326 while (p >= 0 && fh.peek(&ch, 1) == 1 && (ch == '\n' || ch == '\r'))
327 fh.seek(--p);
328 }
329
330 qint64 endPos = p;
331 // loop backward through file until count == maxLines or start of file is reached
332 while (p >= 0) {
333 if (fh.peek(&ch, 1) != 1)
334 break;
335 // is this newline and then did we get enough lines yet?
336 if (ch == '\n') {
337 // check for a CR before of this NL
338 quint8 skipChars = 1;
339 if (fh.seek(p-1) && fh.peek(&ch, 1) == 1) {
340 if (ch == '\r') {
341 --p;
342 skipChars = 2;
343 }
344 else {
345 fh.seek(p);
346 }
347 }
348 // check if this line should be skipped
349 if (skipLines) {
350 --skipLines;
351 endPos = p - 1;
352 //qDebug() << endPos << p << skipLines;
353 }
354 else if (maxLines && ++count == maxLines) {
355 fh.seek(p += skipChars); // don't include the last newline we found
356 --p;
357 break;
358 }
359 }
360 // check the previous character on next iteration.
361 fh.seek(--p);
362 }
363// if (trimTrailing && endPos) {
364// fh.seek(endPos);
365// while (endPos >= 0 && fh.peek(&ch, 1) == 1 && (ch == '\n' || ch == '\r'))
366// fh.seek(--endPos);
367// fh.seek(p);
368// }
369
370 //qDebug() << endPos << p << skipLines << maxLines << count;
371 // Read all bytes from current position to end position
372 if (!skipLines && endPos > p)
373 return fh.read(endPos - p);
374 return QString();
375 }

377 static OpenMode modeToFlags(const QString &mode)
378 {
379 OpenMode f;
380 bool hasA = mode.contains('a');
381 if (mode.contains('+'))
382 f |= O_RDWR;
383 else if (mode.contains('r'))
384 f |= O_RDONLY;
385 if (hasA || f.testFlag(O_WRONLY) || mode.contains('w'))
386 f.setFlag(O_WRONLY).setFlag(hasA ? O_APPEND : O_TRUNC);
387 if (!mode.contains('b'))
388 f |= O_TEXT;
389 if (mode.contains('x'))
390 f |= O_EXCL;
391 else if (mode.contains('n'))
392 f |= O_NOCREAT;
393 if (mode.contains('s'))
394 f |= O_DIRECT;
395 return f;
396 }
